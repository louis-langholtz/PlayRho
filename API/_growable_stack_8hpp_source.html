58  void push(const ElementType& element)
59  {
60  if (m_count == m_capacity)
61  {
62  T* old = m_stack;
63  m_capacity *= GetBufferGrowthRate();
64  m_stack = Alloc<T>(m_capacity);
65  std::memcpy(m_stack, old, m_count * sizeof(T));
66  if (old != m_array)
67  {
68  Free(old);
69  }
70  }
71 
72  m_stack[m_count] = element;
73  ++m_count;
74  }
